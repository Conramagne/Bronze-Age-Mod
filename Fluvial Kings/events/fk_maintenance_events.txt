#Destroys Amazon title if held by non-Amazon (on_new_holder, on_character_convert_culture)

		}
	}

			unsafe_destroy_landed_title = THIS
		}
	}
}

#AI receives event to distribute inherited kingdom (on_new_holder, on_new_holder_usurpation)
character_event = {
	id = fkm.6
	desc = EVTDESCFKM6
	is_triggered_only = yes
	
	# ai = yes
	
	trigger = {
		OR = {
			AND = {
				FROM = {
					tier = COUNT
				}
				tier = DUKE
			}
			AND = {
				FROM = {
					tier = DUKE
				}
				tier = KING
			}
			AND = {
				FROM = {
					tier = KING
				}
				tier = EMPEROR
			}
			AND = {
				FROM = {
					tier = COUNT
					dejure_liege_title = {
						tier = DUKE
						OR = {
							has_holder = no
							NOT = {
								holder = ROOT
							}
							NOR = {
								ROOT = {
									any_realm_lord = {
										has_landed_title = PREVPREV
									}
								}
								ROOT = {
									has_landed_title = PREV
								}
							}
						}
					}
				}
				higher_real_tier_than = COUNT
			}
			AND = {
				FROM = {
					tier = DUKE
					kingdom = {
						OR = {
							has_holder = no
							NOT = {
								holder = ROOT
							}
							NOR = {
								ROOT = {
									any_realm_lord = {
										has_landed_title = PREVPREV
									}
								}
								ROOT = {
									has_landed_title = PREV
								}
							}
						}
					}
				}
				higher_real_tier_than = DUKE
			}
		}
		OR = {
			government = hegemonic_government
			government = military_governorate_government
		}
		NOT = {
			has_character_flag = dont_fire_fkm6
		}
	}
	
	immediate = {
		if = {
			limit = {
				FROM = {
					higher_real_tier_than = COUNT
				}
			}
			FROM = {
				capital_scope = {
					save_event_target_as = capital_county
				}
			}
		}
		else = {
			FROM = {
				save_event_target_as = capital_county
			}
		}
	}
	
	option = {#Give land to player who was child of FROMFROM
		name = EVTOPTEFKM6
		trigger = {
			ai = yes
			FROMFROM = {
				any_dynasty_member = {
					ai = no
					is_adult = yes
				}
			}
		}
		random_dynasty_member = {
			limit = {
				ai = no
				is_adult = yes
			}
			ROOT_FROM = {
				any_de_jure_vassal_title = {
					limit = {
						higher_real_tier_than = BARON
						OR = {
							has_holder = no
							holder = ROOT
						}
					}
					grant_title = PREVPREV
				}
			}
			ROOT_FROM = {
				grant_title = PREV
			}
		}
		ai_chance = {
			factor = 25
			modifier = {
				factor = 2
				trait = arbitrary
			}
			modifier = {
				factor = 0.5
				trait = just
			}
		}
	}
	
	option = {#Creates vassal of your culture
		name = EVTOPTAFKM6
		custom_tooltip = {
			text = EVTOPTAFKM6_TT
		}
		if = {
			limit = {
				culture = amazonian
			}
			create_character = {
				random_traits = yes
				age = 30
				female = yes
				culture = ROOT
				religion = ROOT
				dynasty = random
			}
		}
		else = {
			create_character = {
				random_traits = yes
				age = 30
				female = no
				culture = ROOT
				religion = ROOT
				dynasty = random
			}
		}
		new_character = {
			ROOT_FROM = {
				any_de_jure_vassal_title = {
					limit = {
						higher_real_tier_than = BARON
						OR = {
							has_holder = no
							holder = ROOT
						}
					}
					grant_title = PREVPREV
				}
			}
		}
		new_character = {
			ROOT_FROM = {
				grant_title = PREV
			}
		}
		ai_chance = {
			factor = 25
			modifier = {
				factor = 2
				trait = arbitrary
			}
			modifier = {
				factor = 0.5
				trait = just
			}
		}
	}
	
	option = {#Give land to random close relative
		name = EVTOPTBFKM6
		custom_tooltip = {
			text = EVTOPTAFKM6_TT
		}
		trigger = {
			any_close_relative = {
				is_adult = yes
				at_location = ROOT
				is_landed = no
				OR = {
					AND = {
						culture = amazonian
						culture = ROOT
					}
					female = no
				}
				spouse = {
					is_landed = no
				}
			}
		}
		random_close_relative = {
			limit = {
				is_adult = yes
				at_location = ROOT
				is_landed = no
				OR = {
					AND = {
						culture = amazonian
						culture = ROOT
					}
					female = no
				}
				spouse = {
					is_landed = no
				}
			}
			ROOT_FROM = {
				any_de_jure_vassal_title = {
					limit = {
						higher_real_tier_than = BARON
						OR = {
							has_holder = no
							holder = ROOT
						}
					}
					grant_title = PREVPREV
				}
			}
			ROOT_FROM = {
				grant_title = PREV
			}
		}
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				trait = arbitrary
			}
			modifier = {
				factor = 0.5
				trait = just
			}
		}
	}
		
	option = {#Creates vassal of appropriate culture
		custom_tooltip = {
			text = EVTOPTCFKM6_TT
		}
		name = EVTOPTCFKM6
		trigger = {
			NOT = {
				trait = cruel
			}
			NOT = {
				event_target:capital_county = {
					culture = ROOT
				}
			}
		}
		if = {
			limit = {
				event_target:capital_county = {
					culture = amazonian
				}
			}
			create_character = {
				random_traits = yes
				age = 30
				female = yes
				culture = event_target:capital_county
				religion = event_target:capital_county
				dynasty = random
			}
		}
		else = {
			create_character = {
				random_traits = yes
				age = 30
				female = no
				culture = event_target:capital_county
				religion = event_target:capital_county
				dynasty = random
			}
		}
		new_character = {
			ROOT_FROM = {
				any_de_jure_vassal_title = {
					limit = {
						higher_real_tier_than = BARON
						OR = {
							has_holder = no
							holder = ROOT
						}
					}
					grant_title = PREVPREV
				}
			}
		}
		new_character = {
			ROOT_FROM = {
				grant_title = PREV
			}
		}
		ai_chance = {
			factor = 25
			modifier = {
				factor = 0.5
				trait = arbitrary
			}
			modifier = {
				factor = 3 
				FROMFROM = {
					culture = event_target:capital_county
				}
			}
			modifier = {
				factor = 2
				trait = just
			}
		}
	}

	option = {#Player chooses their own path
		trigger = {
			ai = no
		}
		name = EVTOPTDFKM6
	}
}

#Maintenance to prevent shenaniganery with fkm.6
character_event = {
	id = fkm.9
	hide_window = yes
	is_triggered_only = yes
		
	trigger = {
		higher_real_tier_than = BARON
	}
	
	immediate = {
		any_demesne_title = {
			current_heir = {
				character_event = { id = fkm.10 } #Sets the flag
				character_event = { id = fkm.11 days = 2 } #Clears fkm.9 flag
			}
		}		
	}
}
