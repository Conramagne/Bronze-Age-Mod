character_event = {
	id = fkcg.1
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		has_landed_title = e_hellenic_pantheon
		year = 2561
		NOT = {
			has_global_flag = greek_collapse
		}
	}
	
	immediate = {
		set_global_flag = greek_collapse
		any_province = {
			limit = {
				OR = {
					region = world_greece
					kingdom = {
						title = k_kupros
					}
				}
			}
			add_province_modifier = {
				name = severe_drought
				duration = -1
			}
		}
		any_playable_ruler = {
			limit = {
				any_demesne_province = {
					OR = {
						region = world_greece
						kingdom = {
							title = k_kupros
						}
					}
				}
			}
			narrative_event = { id = fkcg.2 } #Drought strikes Greece
			narrative_event = { id = fkcg.3 days = 100 } #Earthquakes strike Greece
		}
	}
}
