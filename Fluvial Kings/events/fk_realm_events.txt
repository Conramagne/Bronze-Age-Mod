namespace = fkr

#Removes special modifiers for provinces of your culture
character_event = {
	id = fkr.1
	hide_window = yes
	trigger = {
		OR = {
			independent = yes
			in_revolt = yes
		}
		any_realm_province = {
			OR = {
				culture = ROOT
				AND = {
					culture_group = hellene
					culture_group = ROOT
				}
			}
			OR = {
				has_province_modifier = oppressive_rule
				has_province_modifier = autonomous_rule
				has_province_modifier = tax_tribute
				has_province_modifier = levy_tribute
				has_province_modifier = foreign_rule
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		any_realm_province = {
			limit = {
				OR = {
					culture = ROOT
					AND = {
						culture_group = hellene
						culture_group = ROOT
					}
				}
			}
			remove_province_modifier = oppressive_rule
			remove_province_modifier = autonomous_rule
			remove_province_modifier = tax_tribute
			remove_province_modifier = levy_tribute
			remove_province_modifier = foreign_rule
		}
	}
}

#Removes Oppressive Rule if one lacks the law
character_event = {
	id = fkr.2
	hide_window = yes
	trigger = {
		independent = yes
		NOT = {
			has_law = foreigner_law_0
		}
		any_realm_province = {
			has_province_modifier = oppressive_rule
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		any_realm_province = {
			remove_province_modifier = oppressive_rule
		}
		
		if = {
			limit = {
				has_law = foreigner_law_2
			}
			any_realm_province = {
				limit = {
					NOR = {
						culture = ROOT
						AND = {
							culture_group = hellene
							culture_group = ROOT
						}
					}
				}
				add_province_modifier = {
					name = autonomous_rule
					duration = -1
				}
			}
		}
	}
}

#Event to give Viceroy Succession to Military Governorates (on_new_holder, on_startup)
character_event = {
	id = fkr.10
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		OR = {
			government = military_governorate_government
			AND = {
				top_liege = {
					has_law = ze_administration_laws_3
				}
				independent = no
			}
		}
		# NOT = {
			# has_law = succ_viceroyalty
		# }
	}
	
	immediate = {
		if = {
			limit = {
				liege = {
					OR = {
						trait = weak_emperor
						trait = figurehead_emperor
					}
				}
			}
			if = {
				limit = {
					primary_title = {
						tier = DUKE
					}
				}
				set_government_type = feudal_government
			}
			else = {
				set_government_type = regional_kingdom_government
			}
		}
		else_if = {
			limit = {
				primary_title = {
					tier = DUKE
				}
				government = military_governorate_government
			}
			any_demesne_title = {
				limit = {
					tier = DUKE
				}
				add_law = {
					law = succ_viceroyalty
					opinion_effect = no
					cooldown = no
				}
			}
		}
		else_if = {
			limit = {
				primary_title = {
					tier = KING
				}
				government = military_governorate_government
			}
			any_demesne_title = {
				limit = {
					tier = KING
				}
				add_law = {
					law = succ_viceroyalty
					opinion_effect = no
					cooldown = no
				}
			}
		}
		if = {

				}
			}
		}
		else_if = {
			limit = {
				primary_title = {
					tier = DUKE
				}
				AND = {
					top_liege = {
						has_law = ze_administration_laws_3
					}
					independent = no
				}
			}
			any_demesne_title = {
				limit = {
					tier = DUKE
				}
				add_law = {
					law = succ_viceroyalty
					opinion_effect = no
					cooldown = no
				}
			}
		}
		else_if = {
			limit = {
				primary_title = {
					tier = KING
				}

	id = fkr.20
	hide_window = yes
	trigger = {
		liege = {
			has_law = foreigner_law_0
		}
		NOT = {
			has_opinion_modifier = {
				who = liege
				name = opinion_oppressive_overlord
			}
		}
		NOT = {
			culture = liege
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		opinion = {
			modifier = opinion_oppressive_overlord

		}
	}
}

#Setup (on_startup)
character_event = {
	id = fkr.100
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		government = hegemonic_government
		NOT = {
			primary_title = {
				has_title_flag = imperial_authority_set
			}
		}
		NOT = {
			has_character_flag = flag_god
		}
	}
	
	immediate = {
		add_character_modifier = {
			name = imperial_authority
			duration = -1
		}
		if = {
			limit = {
				has_landed_title = e_hittite_empire
			}
			primary_title = {
				set_title_flag = imperial_authority_set
				set_title_flag = imperial_authority_primary_set
			}
			primary_title = {
				set_variable = { which = var_imperial_authority value = 50 }
			}
			set_character_flag = law_no_change
			set_council_law_strong_effect = yes
			clr_character_flag = law_no_change
		}
		else = {
			primary_title = {
				set_title_flag = imperial_authority_set
				set_title_flag = imperial_authority_primary_set
				set_title_flag = flag_hegemonic_government
			}
			primary_title = {
				set_variable = { which = var_imperial_authority value = 50 }
			}
		}
	}
}

	hide_window = yes

	}
