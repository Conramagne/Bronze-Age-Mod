namespace = fkca

character_event = {
	id = fkca.1
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		has_landed_title = e_hellenic_pantheon
		year = 2561
		NOT = {
			has_global_flag = anatolian_collapse
		}
	}
	
	immediate = {
		set_global_flag = anatolian_collapse
		any_province = {
			limit = {
				region = world_asia_minor
			}
			add_province_modifier = {
				name = severe_drought
				duration = -1
			}
		}
		any_playable_ruler = {
			limit = {
				any_demesne_province = {
					region = world_asia_minor
				}
			}
			narrative_event = { id = fkca.2 } #Drought strikes Anatolia
			narrative_event = { id = fkca.3 days = 100 } #Earthquakes strike Anatolia
		}
	}
}

#Drought strikes Anatolia
narrative_event = {
	id = fkca.2
	title = EVTTITLEFKCA2
	desc = EVTDESCFKCA2
	hide_from = yes
	is_triggered_only = yes
	
	option = {
		name = EVTOPTAFKCA2
		if = {
			limit = {
				primary_title = {
					check_variable = { which = palace_bureaucracy_number value = 30 }
				}
			}
			custom_tooltip = {
				text = bureau_decrease_20_TT
				primary_title = {
					change_variable = { which = palace_bureaucracy_number value = -20 }
				}
			}
		}
		set_character_flag = law_no_change
		primary_title = {
			add_law = {
				law = ze_administration_laws_0
				opinion_effect = no
			}
		}
		clr_character_flag = law_no_change
	}
}
