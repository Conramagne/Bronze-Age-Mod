#Puts Twelve Tribes into Alliance (on_startup)
character_event = {
	id = fkc.1
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		has_twelve_tribe_title_trigger = yes
		any_playable_ruler = {
			has_twelve_tribe_title_trigger = yes
			NOR = {
				is_allied_with = ROOT
				character = ROOT
			}
		}
	}
	
	immediate = {
		any_playable_ruler = {
			limit = {
				has_twelve_tribe_title_trigger = yes
				NOR = {
					is_allied_with = ROOT
					character = ROOT
				}
			}
			add_alliance = {
				who = ROOT
				years = 100
			}
			reverse_opinion = {
				modifier = in_non_aggression_pact
				years = 100
				who = ROOT
			}
		}
	}
}

#Israelite Tribe gets attacked
character_event = {
	id = fkc.3
	desc = EVTDESCFKC3
	trigger = {
		has_twelve_tribe_title_trigger = yes
		is_primary_war_defender = yes
		NOR = {
			higher_tier_than = DUKE
			any_playable_ruler = {
				has_landed_title = k_judge_of_israel
			}
		}
		any_war = {
			defender = { character = ROOT }
			attacker = {
				OR = {
					tier = KING
					tier = EMPEROR
				}
			}
		}
		any_playable_ruler = {
			has_twelve_tribe_title_trigger = yes
			NOR = {
				character = ROOT
				higher_tier_than = DUKE
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = { #Begin Judges
		name = EVTOPTAFKC3
		random_character = {
			save_event_target_as = judge_of_israel
			limit = {
				is_landed = no
				is_adult = yes
				is_female = no
				is_incapable = no
				prisoner = no 
				is_inaccessible_trigger = no
				religion = berith
				is_evil_trigger = no
				is_impious_trigger = no
			}
			preferred_limit = {
				is_benevolent_trigger = yes
				is_pious_trigger = yes
				has_education_martial_trigger = yes
			}
			preferred_limit = {
				OR = {
					is_benevolent_trigger = yes
					is_pious_trigger = yes
					has_education_martial_trigger = yes
				}
			}
			grant_title_no_reverse_opinion = k_judge_of_israel
			primary_title = {
				add_law = {
					law = succ_feudal_elective
					reverse_opinion_effect = no
					cooldown = no
				}
			}
			ROOT = {
				set_defacto_liege = PREV
			}
			join_defender_wars = ROOT
			any_playable_ruler = {
				limit = {
					has_twelve_tribe_title_trigger = yes
					NOT = {
						higher_tier_than = DUKE
					}
				}
				character_event = { id = fkc.4 }				
			}
		}
	}
	option = { #Refuse help of Judge
		name = EVTOPTBFKC3
		
		trigger = {
			ai = no
		}
	}
}

#Berith Province Lapses
character_event = {
	id = fkc.8
	desc = EVTDESCFKC8
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		higher_tier_than = BARON
		religion = berith
		any_demesne_province = {
			religion = berith
		}
	}
	weight_multiplier = {
		days = 1 
		modifier = {
			factor = 10
			NOT = { religion_authority = 0.5 }
		}
	}
	immediate = {
		if = {
			limit = {
				capital_scope = {
					religion = berith
				}
			}
			capital_scope = {
				religion = canaanite_religion
			}
			owner = {
				character_event = { id = fkc.9 } #Owner Notified
			}
		}
		else = {
			random_demesne_province = {
				limit = {
					religion = berith
				}
				religion = canaanite_religion
			}
			owner = {
				character_event = { id = fkc.9 } #Owner Notified
			}
		}
	}
}
