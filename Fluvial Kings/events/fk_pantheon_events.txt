	}

						mercenary = no
					}

						mercenary = no
					}

					mercenary = no
				}
				reverse_opinion = { #God's opinion on any_character
					who = ROOT
					value > 0
				}
				NOT = {
					has_opinion_modifier = { #any_character's opinion on God

					}

			}
